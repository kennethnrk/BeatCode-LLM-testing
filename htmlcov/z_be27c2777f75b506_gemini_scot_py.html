3def Solve(satisfaction: list[int]) -> int: 

4 # Step 1: Sort the satisfaction array in ascending order. 

5 # This is crucial for the greedy strategy. By sorting, 

6 # when we iterate from right to left, each 's' we consider 

7 # is the largest available remaining satisfaction value, and 

8 # if added, it will become the smallest element in our chosen subset. 

9 satisfaction.sort() 

10 

11 n = len(satisfaction) 

12 

13 # Step 2: Initialize variables. 

14 # `current_satisfaction_sum` tracks the sum of satisfaction levels 

15 # for all dishes currently included in our optimal subset. 

16 current_satisfaction_sum = 0 

17 

18 # `current_like_time_sum` tracks the total like-time coefficient sum 

19 # for all dishes currently included in our optimal subset. 

20 # Initialized to 0, representing the option of cooking no dishes. 

21 current_like_time_sum = 0 

22 

23 # Step 3 & 4: Iterate from right to left and apply greedy decision. 

24 # Loop from the largest satisfaction level (index n-1) down to the smallest (index 0). 

25 for i in range(n - 1, -1, -1): 

26 s = satisfaction[i] 

27 

28 # Greedy Decision Check: 

29 # If adding the current dish 's' to the existing sum of satisfaction levels 

30 # (current_satisfaction_sum) results in a positive value, it means 

31 # the overall `current_like_time_sum` will increase. 

32 # This is based on the derivation: New_Total_Like_Time = Old_Total_Like_Time + (s + Old_Current_Satisfaction_Sum) 

33 if current_satisfaction_sum + s > 0: 

34 # Step 4b: If beneficial, include this dish. 

35 # Update the sum of satisfaction levels. 's' is now part of the selected dishes. 

36 current_satisfaction_sum += s 

37 

38 # Update the total like-time coefficient sum. 

39 # The total sum increases by the new `current_satisfaction_sum` 

40 # (which is `s` plus the sum of previously chosen dishes). 

41 current_like_time_sum += current_satisfaction_sum 

42 else: 

43 # Step 4c: If `current_satisfaction_sum + s <= 0`, adding 's' would not 

44 # increase the total like-time sum. Since 's' is the largest 

45 # of the remaining unchosen dishes (due to right-to-left iteration), 

46 # adding any smaller dish would also not be beneficial. 

47 # Therefore, we break the loop as we've found our maximum. 

48 break 

49 

50 # Step 5: Return `current_like_time_sum`. 

51 # At this point, `current_like_time_sum` holds the maximum possible 

52 # like-time coefficient sum. If no dishes were chosen (e.g., all 

53 # satisfaction levels were negative and the loop broke immediately), 

54 # it correctly remains 0. 

55 return current_like_time_sum
