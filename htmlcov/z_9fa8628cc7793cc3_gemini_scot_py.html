3def Solve(envelopes: list[list[int]]) -> int: 

4 # Handle the edge case where no envelopes are provided. 

5 if not envelopes: 

6 return 0 

7 

8 # Step 1: Sort the envelopes 

9 # Sort primarily by width (x[0]) in ascending order. 

10 # If widths are equal, sort by height (-x[1]) in descending order. 

11 # This ensures that envelopes with the same width cannot be part of the same 

12 # increasing subsequence of heights, effectively handling the strict width condition. 

13 envelopes.sort(key=lambda x: (x[0], -x[1])) 

14 

15 # Step 2: Extract heights 

16 # Create a list containing only the heights from the sorted envelopes. 

17 heights = [env[1] for env in envelopes] 

18 

19 # Step 3: Find the Longest Increasing Subsequence (LIS) of the heights 

20 # 'tails' will store the smallest ending element of all increasing subsequences 

21 # of length (index + 1) found so far. 

22 tails = [] 

23 

24 for h in heights: 

25 # Use bisect_left to find the index 'idx' where 'h' can be inserted 

26 # in 'tails' to maintain sorted order. 

27 # 'idx' will be the first index such that tails[idx] >= h. 

28 idx = bisect.bisect_left(tails, h) 

29 

30 # If 'idx' is equal to the current length of 'tails', it means 'h' is 

31 # strictly greater than all elements currently in 'tails'. 

32 # So, we extend the longest increasing subsequence found so far by appending 'h'. 

33 if idx == len(tails): 

34 tails.append(h) 

35 # Otherwise, 'h' is not strictly greater than all elements. 

36 # We found an increasing subsequence of length 'idx + 1' ending with tails[idx]. 

37 # By replacing tails[idx] with 'h', we form an increasing subsequence of 

38 # the same length 'idx + 1' but with a potentially smaller tail 'h'. 

39 # A smaller tail is more favorable as it allows more subsequent elements 

40 # to extend this subsequence further. 

41 else: 

42 tails[idx] = h 

43 

44 # The final length of the 'tails' list represents the length of the LIS, 

45 # which is the maximum number of envelopes that can be Russian dolled. 

46 return len(tails)
